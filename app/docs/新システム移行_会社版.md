# 株式AI システム移行計画書（会社版）

## 概要

Flask + Jinja2 + SQLite のローカルシステムから、モダンなクラウドネイティブ構成への移行計画。
会社での利用を前提とし、将来の商用化も視野に入れた設計。

### 移行のゴール

1. **会社メンバーがどこからでもアクセス可能**
2. **毎朝Slackにシグナルが届く**
3. **高速なレスポンス**（DBから読む）
4. **必要な時はリアルタイムデータ取得可能**
5. **運用コスト最小**（月額$0〜45）

## 1. 現状の問題点

### 1.1 技術的負債

| 現在 | 問題点 | 影響 |
|------|--------|------|
| **Jinja2** | サーバーで毎回HTML生成 | 遅い、スケールしない |
| **SQLite** | ファイルベース、同時書き込み不可 | マルチユーザー不可 |
| **Flask dev server** | 本番向けでない | 不安定 |
| **yfinance毎回呼び出し** | ボタン押すたびにAPI | レート制限、遅い |
| **ローカル実行** | 自分のPCでしか動かない | 会社で共有不可 |

### 1.2 現在のアーキテクチャ


### 5.2 シグナル計算スクリプト

```python
# scripts/signal_runner.py
"""
毎日のシグナル計算 + Slack通知
"""
import os
import json
from datetime import datetime, timedelta
from typing import List, Dict, Any

import yfinance as yf
import pandas as pd
from supabase import create_client, Client

# 環境変数
SUPABASE_URL = os.environ['SUPABASE_URL']
SUPABASE_KEY = os.environ['SUPABASE_SERVICE_KEY']
SLACK_WEBHOOK = os.environ.get('SLACK_WEBHOOK_URL')

# Supabaseクライアント
supabase: Client = create_client(SUPABASE_URL, SUPABASE_KEY)

# 監視銘柄リスト
WATCHLIST = [
    # Penny (<$20)
    'SOUN', 'RGTI', 'IONQ', 'SOFI', 'GSAT', 'MARA', 'PLUG', 'NIO',
    # Mid ($20-100)
    'COIN', 'RKLB', 'PLTR', 'HOOD', 'AFRM', 'SNAP', 'ROKU',
    # Large ($100+)
    'NVDA', 'TSLA', 'META', 'AAPL', 'MSFT', 'GOOGL', 'AMZN', 'AMD'
]


def get_market_regime(spy_data: pd.DataFrame) -> Dict[str, Any]:
    """市場レジームを判定"""
    latest = spy_data.iloc[-1]
    ema_200 = spy_data['Close'].ewm(span=200, adjust=False).mean().iloc[-1]
    ema_21 = spy_data['Close'].ewm(span=21, adjust=False).mean()
    ema_21_slope = ema_21.iloc[-1] - ema_21.iloc[-6]  # 5日間の変化

    above_200 = latest['Close'] > ema_200
    ema21_up = ema_21_slope > 0

    if above_200 and ema21_up:
        regime = 'BULL'
    elif above_200 and not ema21_up:
        regime = 'WEAKENING'
    elif not above_200 and ema21_up:
        regime = 'RECOVERY'
    else:
        regime = 'BEAR'

    return {
        'regime': regime,
        'spy_close': float(latest['Close']),
        'spy_ema_200': float(ema_200),
        'spy_ema_21': float(ema_21.iloc[-1]),
        'spy_ema_21_slope': float(ema_21_slope)
    }


def calculate_signals(ticker: str, data: pd.DataFrame, regime: str) -> Dict[str, Any]:

def save_to_supabase(signals: List[Dict[str, Any]], regime_data: Dict[str, Any]):
    """結果をSupabaseに保存"""
    today = datetime.now().strftime('%Y-%m-%d')

    # 市場レジーム保存
    supabase.table('market_regime').upsert({
        'date': today,
        **regime_data
    }, on_conflict='date').execute()

    # シグナル保存
    for signal in signals:
        supabase.table('signals').upsert(
            signal,
            on_conflict='ticker,date'
        ).execute()


def send_slack_notification(signals: List[Dict], regime: Dict):
```

```typescript
// lib/api/stocks.ts
// オンデマンドキャッシュ方式の株価取得

import { createClient } from '@/lib/supabase/client'

const CACHE_DURATION_MS = 5 * 60 * 1000  // 5分

export async function getStockPrice(ticker: string) {
  const supabase = createClient()
  const today = new Date().toISOString().split('T')[0]

  // 1. DBからキャッシュを確認
  const { data: cached } = await supabase
    .from('stock_prices')
    .select('*')
    .eq('ticker', ticker)
    .eq('date', today)
    .single()

  // 2. キャッシュが新しければそのまま返す
  if (cached) {
    const updatedAt = new Date(cached.updated_at).getTime()
    const now = Date.now()

    if (now - updatedAt < CACHE_DURATION_MS) {
      return { data: cached, source: 'cache' }
    }
  }

  // 3. キャッシュが古いor存在しない → API取得
  const response = await fetch(`/api/stock/${ticker}/refresh`, {
    method: 'POST'
  })

  if (!response.ok) {
    // APIエラー時は古いキャッシュを返す（あれば）
    if (cached) return { data: cached, source: 'stale-cache' }
    throw new Error('Failed to fetch stock price')
  }

  const { data } = await response.json()
  return { data, source: 'api' }
}
```

```typescript
// app/api/stock/[ticker]/refresh/route.ts
// 強制リフレッシュ（APIから取得してDB保存）

import { NextRequest, NextResponse } from 'next/server'
import { createClient } from '@/lib/supabase/server'
import yahooFinance from 'yahoo-finance2'

export async function POST(
  request: NextRequest,
  { params }: { params: { ticker: string } }
) {
  const { ticker } = params
  const supabase = createClient()

  try {
    // Yahoo Financeから最新データ取得
    const quote = await yahooFinance.quote(ticker)

    const priceData = {
      ticker,
      date: new Date().toISOString().split('T')[0],
      open: quote.regularMarketOpen,
      high: quote.regularMarketDayHigh,
      low: quote.regularMarketDayLow,
      close: quote.regularMarketPrice,
      volume: quote.regularMarketVolume,
      updated_at: new Date().toISOString()
    }

    // Supabaseに保存（次回以降のキャッシュになる）
    const { error } = await supabase
      .from('stock_prices')
      .upsert(priceData, { onConflict: 'ticker,date' })

    if (error) throw error

    return NextResponse.json({
      success: true,
      data: priceData,
      source: 'api'
    })

  } catch (error) {
    return NextResponse.json(
      { success: false, error: String(error) },
      { status: 500 }
    )
  }
}
```
